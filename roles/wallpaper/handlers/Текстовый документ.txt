---
- name: Restart GNOME
  ansible.builtin.command: >
    gnome-shell --replace &

- name: Restart MATE
  ansible.builtin.command: >
    mate-session-save --logout &

- name: Restart KDE
  ansible.builtin.command: >
    qdbus org.kde.ksmserver /KSMServer logout 0 0 0

- name: Restart XFCE
  ansible.builtin.command: >
    xfce4-session-logout --logout &

- name: Restart LXDE
  ansible.builtin.command: >
    lxsession-logout
